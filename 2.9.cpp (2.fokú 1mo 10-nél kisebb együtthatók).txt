#include <iostream>
#include <cmath>

using namespace std;

// 2.9) Sorold fel azokat a másodfokú egyenleteket, amelyek minden együtthatója 0 és 10 közötti egész szám, és pontosan egy megoldása van.

int main()
{
    int a = 1, b = 1, c = 1;
    for (int c = 1; c < 10; c++)
    {
        for (int b = 1; b < 10; b++)
        {
            for (int a = 1; a < 10; a++)
            {
                if (pow(b, 2)-4*a*c == 0)
                {
                    cout << a << "x^2 + " << b << "x + " << c << endl;
                }
            }
        }
    }
    return 0;
}
