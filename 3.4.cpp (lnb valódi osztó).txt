#include <iostream>

using namespace std;

// 3.4) Add meg egy természetes szám legnagyobb valódi osztóját.

int main()
{
    int a, maxi = 0;
    cin >> a;
    for (int i=1; i<a; i++)
    {
        if (a%i == 0)
        {
            maxi = i;
        }
    }
    cout << maxi;
    return 0;
}
