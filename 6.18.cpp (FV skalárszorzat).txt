#include <iostream>

using namespace std;

// 6.18) Valósítsd meg a skalárszorzat mûveletet függvényként

int skalarszorzat (int n)
{
    int a, b, sum;
    for (int i = 0; i < n; i++)
    {
        cin >> a >> b;
        sum += a*b;
    }
    return sum;
}

int main()
{
    int n;
    cin >> n;
    cout << skalarszorzat (n);
    return 0;
}
