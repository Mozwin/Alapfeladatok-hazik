#include <iostream>
#include <vector>

using namespace std;

// 6.17) Valósíts meg függvényt, ami mátrixot kap paraméterül, és visszaadja a mátrix transzponáltját

void transz (int n, int m)
{
    int a[n][m];
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            cin >> a[i][j];
        }
    }
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            cout << a[j][i] << ' ';
        }
        cout << endl;
    }
}
int main()
{
    int n, m;
    cin >> n >> m;
    transz (n, m);
    return 0;
}
