#include <iostream>

using namespace std;

// 3.6a) Sorold fel az elsõ N tökéletes számot (olyat természetes számot, ami megegyezik magánál kisebb osztói összegével).

int main()
{
    int n;
    cin >> n;
    int a = 1;
    int db = 0;
    int sum;

    while (db < n)
    {
        sum = 0;
        for (int i = 1; i<a; i++)
        {
            if (a % i == 0)
            {
                sum += i;
            }
        }
        if (sum ==  a)
        {
            cout << a << ' ';
            db += 1;
        }
        a += 1;
    }

    return 0;
}
