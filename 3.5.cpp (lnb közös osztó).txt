#include <iostream>

using namespace std;

// 3.5) Add meg 2 természetes szám legnagyobb közös osztóját.

int main()
{
    int a,b, maxi = 0;
    cin >> a;
    cin >> b;
    for (int i=1; i<=b && i <= a ; i++)
    {
        if (a%i == 0 && b%i == 0 && i > maxi)
        {
            maxi = i;
        }
    }
    cout << maxi;
    return 0;
}
