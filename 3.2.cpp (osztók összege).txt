#include <iostream>

using namespace std;

// 3.2) Add meg egy természetes szám osztóinak összegét.

int main()
{
    int a;
    cin >> a;
    int sum = 0;

    for (int i = 1; i <= a; i++)
    {
        if (a % i == 0)
        {
            sum += i;
        }
    }
    cout << sum;
    return 0;
}
