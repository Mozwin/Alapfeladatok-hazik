#include <iostream>
#include <fstream>

using namespace std;

// 6.11) Valósítsd meg a void alahuzvakiir(string s) függvényt, ami a paraméterül kapott szöveget új sorba kiírja, és "=" karakterekkel aláhúzza.

void alahuzvakiir (string s)
{
    cout << endl << s << endl;
    for (int i = 0; i<s.length(); i++)
    {
        cout << '=';
    }
    return;
}

int main()
{
    string szoveg;
    cin >> szoveg;
    alahuzvakiir(szoveg);
    return 0;
}
