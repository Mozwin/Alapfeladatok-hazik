#include <iostream>
#include <time.h>
#include <cstdlib>

using namespace std;

// 5.4) Van-e két egyforma elem a vektorban?

int main()
{
    srand (time(0));
    int a;
    cin >> a;
    int b[10];
    bool van = false;

    for (int i=0; i<a; i++)
    {
        b[i] = rand() % 10;
        cout << b[i] << ' ';
    }
    cout << endl;

    for (int i=0; i<a; i++)
    {
        for (int j=i+1; j<a; j++)
        {
            if (b[i] == b[j])
            {
                van = true;
            }

        }
    }
    cout << boolalpha << van << noboolalpha;
    return 0;
}
