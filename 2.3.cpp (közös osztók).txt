#include <iostream>

using namespace std;

// 2.3) Sorold fel két pozitív egész szám közös osztóit.

int main()
{
    int a,b;
    cin >> a;
    cin >> b;
    for (int i = 1; i<=a && i<=b; i++)
    {
        if (a % i == 0 && b % i == 0)
        {
            cout << i << ' ';
        }
    }

    return 0;
}
