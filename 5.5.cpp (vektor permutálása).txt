#include <iostream>
#include <time.h>
#include <cstdlib>

using namespace std;

// 5.5) Vektor permutálása véletlenszerûen.

int main()
{
    srand (time(0));
    int a[10];
    int p[10];
    int r;

    for (int i=0; i<10; i++)
    {
        a[i] = rand() % 10;
        cout << a[i] << ' ' ;
    }
    cout << endl;

    for (int i=0; i<10; i++)
    {
        r = rand() % (10-i);
        p[i] = a[r];
        for (int j=r; j<9-i; j++)
        {
            a[j] = a[j+1];
        }
        cout << p[i] << ' ';
    }
    return 0;
}
