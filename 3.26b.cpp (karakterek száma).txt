#include <iostream>

using namespace std;

// 3.26b) Egy tetszõleges szövegben add meg a karakteret számát.

int main()
{
    string a;
    int sum1 = 0, sum2 = 0;
    getline (cin,a);
    while (a.length() > 0)
    {
        sum1 += 1;
        sum2 = sum2 + a.length ();
        getline(cin, a);
    }
    cout << sum1 << ' ' << sum2;
    return 0;
}
