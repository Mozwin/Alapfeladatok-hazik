#include <iostream>
#include <vector>

using namespace std;

// 6.7) Valósítsd meg a double atlag(vector<double> v) függvényt, ami a paraméterül kapott vektor átlagát adja vissza

double atlag (vector<double> v)
{
    double avg = 0;
    for (unsigned i=0; i<v.size(); i++)
    {
        avg += v[i];
    }
    return (avg/v.size());
}

int main()
{
    vector <double> v = {23, 34, 65, 55};
    cout << atlag(v);
    return 0;
}
