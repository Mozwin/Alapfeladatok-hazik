#include <iostream>

using namespace std;

// 3.10) Add meg egy egész számsorról, hogy hány eleme nagyobb ill. kisebb az átlagnál.

int main()
{
    int a;
    cin >> a;

    double b[a];
    double avg = 0, kb = 0, nb = 0;
    for (int i = 0; i<a; i++)
    {
        cin >> b[i];
        avg += b[i];
    }

    avg = avg / a;

    for (int i=0; i<a; i++)
    {
        if (b[i] < avg)
        {
            kb += 1;
        }
        else if (b[i] > avg)
        {
            nb += 1;
        }
    }
    cout << kb << endl;
    cout << nb;
    return 0;
}
