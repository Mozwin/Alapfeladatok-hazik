#include <iostream>

using namespace std;

// 3.6b) Sorold fel a K-nál kisebb tökéletes számokat.

int main()
{
    int k;
    cin >> k;
    int a = 1;
    int sum;

    while (a < k)
    {
        sum = 0;
        for (int i = 1; i<a; i++)
        {
            if (a % i == 0)
            {
                sum += i;
            }
        }
        if (sum ==  a)
        {
            cout << a << ' ';
        }
        a += 1;
    }

    return 0;
}
