#include <iostream>
#include <fstream>

using namespace std;

// 7.14)  Képfájl

int main()
{
    ofstream KI;    // Logikai fájlnév létrehozása
    KI.open("kis.kep");    // Fizikai fájlnév hozzárendelése a logikai fájlnévhez. Ezt fogjuk kifájl esetén létrehozni. Jelen esetben a kiterjesztése és a neve is bármi lehet.
    KI << 400 << endl;  // Mivel kifájl, ezért csak kiíratni tudunk! <<
    KI << 400 << endl;
    for (int i=0; i<400; i++)
    {
        for (int j=0; j <400; j++)
        {
           KI << i/4 << " " << i/2 << " " << i/6 << " "; // Itt 3 változó kell legyen, amik értéke 0-255 lehet
        }
    }
    KI.close();    // Lezárjuk a file-t, hogy meg lehessen nyitni más programmal
    return 0;
}
